-- 查询
SELECT * FROM user_profile;

-- 查询多列
SELECT device_id,gender,age,university FROM user_profile;

-- 查询结果去重
SELECT DISTINCT university FROM user_profile;

-- 查询结果限制返回行数
SELECT device_id FROM user_profile LIMIT 2;

-- 将查询后的列重新命名
SELECT device_id AS user_infos_example
FROM user_profile LIMIT 2；

-- 查找除复旦大学的用户信息
SELECT device_id,gender,age,university
FROM user_profile
WHERE university <> "复旦大学";

-- 用where过滤空值
SELECT device_id,gender,age,university
FROM user_profile
WHERE age IS NOT NULL;

-- 注意IS和=
SELECT device_id,gender,age,university,gpa
FROM user_profile
WHERE gender = 'male' AND gpa > 3.5;

-- Where in 和Not in
SELECT device_id,gender,age,university,gpa
FROM user_profile
WHERE university IN ('北京大学','复旦大学','山东大学')
AND university IS NOT NULL;

-- 混合
SELECT device_id,gender,age,university,gpa
FROM user_profile
WHERE (gpa>3.5 AND university = '山东大学')
OR (gpa>3.8 AND university = '复旦大学');

-- 查看学校名称中含北京的用户
SELECT device_id,age,university
FROM user_profile
WHERE university LIKE '%北京%';

-- 查找GPA最高值
SELECT MAX(gpa) AS gpa
FROM user_profile
WHERE university = '复旦大学';

-- 计算男生人数以及平均GPA（括号不能有空格）
SELECT
COUNT(*) AS male_num,
AVG(gpa) AS avg_gpa
FROM user_profile
WHERE gender = 'male';

-- 分组计算练习题
SELECT
    gender,
    university,
    COUNT(*) AS user_num,
    ROUND(AVG(active_days_within_30),1)  AS avg_active_day,
    ROUND(AVG(question_cnt),1) AS avg_question_cnt
FROM user_profile
GROUP BY gender,university
ORDER BY gender,university;

-- 分组过滤（注意逗号和having/where）
SELECT
    university,
    AVG(question_cnt) AS avg_question_cnt,
    AVG(answer_cnt) AS avg_answer_cnt
FROM user_profile
GROUP BY university
HAVING avg_question_cnt < 5 OR avg_answer_cnt < 2

-- 分组排序
SELECT
    university,
    ROUND(AVG(question_cnt),4) AS avg_question_cnt
FROM user_profile
GROUP BY university
ORDER BY avg_question_cnt;

-- 多表查询
SELECT
    q.device_id,q.question_id,q.result
FROM question_practice_detail q
JOIN user_profile u
ON u.device_id = q.device_id
WHERE u.university = '浙江大学'

-- 复杂多表查询
SELECT
    u.university,
    ROUND(COUNT(q.question_id)/COUNT(DISTINCT q.device_id),4) AS avg_answer_cnt
FROM user_profile u
JOIN question_practice_detail q
ON u.device_id = q.device_id
GROUP BY university
ORDER BY university

-- 更多的表，理解业务
SELECT
    u.university,
    qd.difficult_level,
    ROUND(COUNT(q.question_id)/COUNT(DISTINCT q.device_id),4)
FROM user_profile u
JOIN question_practice_detail q
    ON u.device_id = q.device_id
JOIN question_detail qd
    ON q.question_id = qd.question_id
GROUP BY u.university,qd.difficult_level

-- 两次输出
SELECT device_id,gender,age,gpa
FROM user_profile
WHERE university = '山东大学'
UNION ALL
SELECT device_id,gender,age,gpa
FROM user_profile
WHERE GENDER = 'male'

-- 条件分组（注意逗号）
SELECT
    CASE
        WHEN age<25 OR age IS NULL THEN '25岁以下'
        ELSE '25岁及以上'
    END AS age_cut,
    COUNT(*) AS number
FROM user_profile
GROUP BY age_cut;

-- 日期分组
SELECT
    DAY(date) AS day,
    COUNT(question_id) AS question_cnt
FROM question_practice_detail
WHERE YEAR(date)=2021 AND MONTH(date) = 8
GROUP BY day;

-- 左连接
  SELECT
    ROUND(AVG(IF(b.device_id IS NOT NULL,1,0)),4) AS avg_ret
FROM (
    SELECT DISTINCT device_id,date
    FROM question_practice_detail
) a
LEFT JOIN (
    SELECT DISTINCT device_id,date
    FROM question_practice_detail
) b
ON a.device_id = b.device_id
    AND DATE_ADD(a.date, INTERVAL 1 DAY) = b.date

-- 截取
SELECT
    SUBSTRING_INDEX(profile,',',-1) AS gender,
    COUNT(*) AS number
FROM user_submit
GROUP BY gender;

SELECT cust_name FROM Customers ORDER BY cust_name DESC

    order_num,prod_id,quantity

SELECT vend_name

-- LIKE 否定
SELECT
   prod_name, prod_desc
FROM Products
WHERE prod_desc NOT LIKE '%toy%'
ORDER BY prod_name;

-- 同一个字段重复写条件，不能直接跟 LIKE，必须带上字段名
SELECT
   prod_name, prod_desc
FROM Products
WHERE prod_desc LIKE '%toy%'
    AND prod_desc LIKE '%carrots%';
    
-- LIKE 先后顺序
SELECT
   prod_name, prod_desc
FROM Products
WHERE prod_desc LIKE '%toy%carrots%'; 

-- 拼接字符串 大写
SELECT
    cust_id,
    cust_name,
    UPPER(CONCAT(
        SUBSTRING(cust_contact,1,2),
        SUBSTRING(cust_city,1,3)
     )) AS user_login
FROM Customers;

-- LIMIT不能写在字段前面 ORDER BY prod_price DESC -- 必须排序，才能取到最大 LIMIT 1
SELECT
    MAX(prod_price) AS max_price
FROM Products
WHERE prod_price<=10

-- SELECT → FROM → WHERE → GROUP BY → HAVING → ORDER BY → LIMIT
SELECT
    order_num,
    COUNT(order_num) AS order_lines
FROM OrderItems
GROUP BY order_num
ORDER BY order_lines

-- HAVING（过滤分组后数据） 与 GROUP BY 的关系
SELECT
    order_num
FROM OrderItems
GROUP BY order_num
HAVING SUM(quantity) >= 100
ORDER BY order_num

-- 内连接（两边都存在）
SELECT
    c.cust_name,
    o.order_num
FROM Customers c
INNER JOIN Orders o
ON c.cust_id = o.cust_id
ORDER BY c.cust_name,o.order_num

-- 严格sql语法
SELECT
  c.cust_name,
  o.order_num,
  SUM(oi.quantity*oi.item_price) AS OrderTotal
FROM Customers c
INNER JOIN Orders o
ON c.cust_id = o.cust_id
INNER JOIN OrderItems oi
ON o.order_num = oi.order_num
GROUP BY c.cust_name,o.order_num
ORDER BY c.cust_name,o.order_num

-- 简单等值连接语法
SELECT
    o.cust_id,
    o.order_date
FROM Orders o, OrderItems oi
WHERE o.order_num = oi.order_num
AND oi.prod_id ='BR01'
ORDER BY o.order_date

-- SQL执行顺序：FROM / JOIN；WHERE；GROUP BY；HAVING；SELECT（计算别名、SUM、ROUND 都在这里！）；ORDER BY
SELECT
    c.cust_name,
    ROUND(SUM(oi.item_price*oi.quantity),3) AS total_price
FROM Customers c
INNER JOIN Orders o
ON c.cust_id = o.cust_id
INNER JOIN OrderItems oi
ON oi.order_num = o.order_num
GROUP BY oi.order_num,c.cust_name
HAVING SUM(oi.item_price*oi.quantity)>=1000
ORDER BY total_price

-- 不能单独写OUTER JOIN
SELECT
    p.prod_name,
    o.order_num
FROM Products p
LEFT OUTER JOIN OrderItems o
ON p.prod_id = o.prod_id
ORDER BY p.prod_name

-- ON 不要漏写
SELECT
    p.prod_name,
    COUNT(o.order_num) AS orders
FROM Products p
LEFT OUTER JOIN OrderItems o
ON o.prod_id = p.prod_id
GROUP BY p.prod_name,p.prod_id
ORDER BY p.prod_name

-- GROUP BY不要漏写
SELECT
    v.vend_id,
    COUNT(p.prod_id) AS prod_id
FROM Vendors v
LEFT JOIN Products p
ON v.vend_id = p.vend_id
GROUP BY v.vend_id
ORDER BY v.vend_id

UNION

SELECT prod_id,quantity

ORDER BY prod_id

SELECT
    prod_id,quantity
FROM OrderItems
WHERE quantity = 100 OR prod_id LIKE 'BNBG%'
ORDER BY prod_id

SELECT prod_name FROM Products

SELECT cust_name FROM Customers
